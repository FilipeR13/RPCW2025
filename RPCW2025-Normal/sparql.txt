. Qual a distribuição de aprendizes por disciplina? (Resultado será uma lista de pares: Disciplina, número de alunos)

PREFIX : <http://www.semanticweb.org/josérodrigues/ontologies/2025/academia/>
SELECT ?nome_disciplina (COUNT(?aprendiz) AS ?num_alunos) {
    ?aprendiz a :Aprendiz ;
              :aprende ?disciplina .
    ?disciplina :nome ?nome_disciplina .
} GROUP BY ?nome_disciplina

. Quais os títulos das obras associadas a cada conceito? (Resultado será uma lista de pares: Conceito, Lista de títulos de obras)

PREFIX : <http://www.semanticweb.org/josérodrigues/ontologies/2025/academia/>
SELECT ?conceito_nome (GROUP_CONCAT(?titulo_obra; SEPARATOR=", ") AS ?titulos) {
    ?conceito a :Conceito ;
              :nome ?conceito_nome .
    ?obra a :Obra ;
          :explica ?conceito ;
          :título ?titulo_obra .
} GROUP BY ?conceito_nome


. Quais os aprendizes que frequentam cada uma das disciplinas? (Resultado será uma lista de pares: Disciplina, Lista dos nomes dos aprendizes que frequentam a disciplina)

PREFIX : <http://www.semanticweb.org/josérodrigues/ontologies/2025/academia/>
SELECT ?disciplina_nome (GROUP_CONCAT(?aprendiz_nome; SEPARATOR=", ") AS ?aprendizes) {
    ?disciplina a :DIsciplina ;
               :nome ?disciplina_nome .
    ?aprendiz a :Aprendiz ;
              :aprende ?disciplina ;
              :nome ?aprendiz_nome .
} GROUP BY ?disciplina_nome

. Qual a distribuição de aprendizes por escalões etários? (Resultado será uma lista de pares: Escalão etário, número aprendizes nesse escalão); Considera os seguintes escalões: [17, 34], [35, 44], [45, 54], [55, 64], [65, 74], [75, 100].

PREFIX : <http://www.semanticweb.org/josérodrigues/ontologies/2025/academia/>
SELECT ?escalao (COUNT(?aprendiz) AS ?numeroDeAprendizes)
WHERE {
  ?aprendiz a :Aprendiz ;
            :idade ?idade .
  BIND(
    IF(?idade >= 17 && ?idade <= 34, "[17, 34]",
    IF(?idade >= 35 && ?idade <= 44, "[35, 44]",
    IF(?idade >= 45 && ?idade <= 54, "[45, 54]",
    IF(?idade >= 55 && ?idade <= 64, "[55, 64]",
    IF(?idade >= 65 && ?idade <= 74, "[65, 74]",
    IF(?idade >= 75 && ?idade <= 100, "[75, 100]", "Fora do Escalão")))))) AS ?escalao)
}
GROUP BY ?escalao
